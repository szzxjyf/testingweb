import pandas as pd

def read_csv_auto_encoding(file_path):
    """尝试多种编码读取 CSV 文件，返回 (DataFrame, encoding)"""
    for enc in ["utf-8-sig", "utf-8", "gb18030", "gbk"]:
        try:
            return pd.read_csv(
                file_path,
                encoding=enc,
                low_memory=False,
                error_bad_lines=False
            ), enc
        except Exception:
            continue

    raise ValueError(f"无法解码 CSV 文件: {file_path}")
